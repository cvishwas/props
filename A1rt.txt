public UserDto saveUser(UserDto user) {
    String url = REMOTE_BASE_URL + "/users";

    HttpHeaders headers = new HttpHeaders();
    headers.setContentType(MediaType.APPLICATION_JSON);
    HttpEntity<UserDto> request = new HttpEntity<>(user, headers);

    ResponseEntity<EtfApiResponse<UserDto>> response =
        restTemplate.exchange(
            url,
            HttpMethod.POST,
            request,
            new ParameterizedTypeReference<EtfApiResponse<UserDto>>() {}
        );

    EtfApiResponse<UserDto> body = response.getBody();
    return body != null ? body.getData() : null;
}
